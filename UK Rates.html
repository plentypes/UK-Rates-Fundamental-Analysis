import pandas as pd

def align_ficc_sheet(file_path):
    # 1. Raw data load
    raw_df = pd.read_excel('rate data.xlsx')

    # 2. Extract and clean 4 distinct series
    df_rate = raw_df[['Rate Date', 'Rate %']].dropna().rename(columns={'Rate Date': 'Date', 'Rate %': 'Base_Rate'})
    df_2y = raw_df[['2y_Date', '2y_Price']].dropna().rename(columns={'2y_Date': 'Date', '2y_Price': '2Y_Yield'})
    df_10y = raw_df[['10y_Date', '10y_Price']].dropna().rename(columns={'10y_Date': 'Date', '10y_Price': '10Y_Yield'})
    df_cpi = raw_df[['CPI_Month/Year', 'CPI']].dropna().rename(columns={'CPI_Month/Year': 'Date', 'CPI': 'CPI_YoY'})

    # 3. Standardize all 'Date' columns
    dfs = [df_rate, df_2y, df_10y, df_cpi]
    for df in dfs:
        df['Date'] = pd.to_datetime(df['Date'])
        df.set_index('Date', inplace=True)
        df.sort_index(inplace=True)

    # 4. Join all together (Outer Join)
    # This creates a 'Master' timeline containing every unique date from all 4 series
    master = df_2y.join([df_10y, df_rate, df_cpi], how='outer')

    # 5. Forward Fill (ffill)
    # This carries the last known Base Rate and CPI value forward to match daily bond prices
    master[['Base_Rate', 'CPI_YoY']] = master[['Base_Rate', 'CPI_YoY']].ffill()

    # 6. Final Clean
    # Drop rows that don't have yield data (where only CPI was recorded)
    master = master.dropna(subset=['2Y_Yield', '10Y_Yield'])
    
    # 7. Add Analysis Columns
    master['Spread_2s10s'] = master['10Y_Yield'] - master['2Y_Yield']

    return master

final_df = align_ficc_sheet('rate data.xlsx')

import matplotlib.pyplot as plt
import matplotlib.gridspec as gridspec

def plot_comprehensive_economic_analysis(df):
    plt.rcParams['axes.facecolor'] = '#FDFDFD'
    fig = plt.figure(figsize=(14, 10))
    gs = gridspec.GridSpec(2, 1, height_ratios=[2, 1]) 
    
    ax1 = plt.subplot(gs[0])
    
    # 1. Gilt Yields
    ax1.plot(df.index, df['2Y_Yield'], label='2Y Gilt Yield', color='#34495e', lw=1.5, alpha=0.9)
    ax1.plot(df.index, df['10Y_Yield'], label='10Y Gilt Yield', color='#e74c3c', lw=1.5, alpha=0.9)
    
    # 2. Plot BoE Base Rate (using step to show discrete policy changes)
    ax1.step(df.index, df['Base_Rate'], label='BoE Base Rate', color='#2980b9', lw=2.5, where='post')
    
    ax1.set_ylabel('Interest Rate / Yield (%)', fontsize=12, fontweight='bold')
    ax1.set_title('UK Economic Fundamental Overlay (2020-2026)', fontsize=15, pad=15)
    ax1.grid(True, alpha=0.2, linestyle='--')

    # 3. Create twin axis for CPI YoY % (Right Axis)
    ax1_cpi = ax1.twinx()
    ax1_cpi.plot(df.index, df['CPI_YoY'], label='CPI YoY Inflation', color='#f39c12', lw=2, linestyle=':', alpha=0.8)
    ax1_cpi.set_ylabel('Inflation Rate (YoY %)', color='#f39c12', fontsize=12, fontweight='bold')
    ax1_cpi.tick_params(axis='y', labelcolor='#f39c12')

    lines1, labels1 = ax1.get_legend_handles_labels()
    lines2, labels2 = ax1_cpi.get_legend_handles_labels()
    ax1.legend(lines1 + lines2, labels1 + labels2, loc='upper left', frameon=True, shadow=True)

    #The Yield Spread (Market Sentiment)
    ax2 = plt.subplot(gs[1], sharex=ax1)
    
    # Using Spread column
    ax2.fill_between(df.index, df['Spread_2s10s'], 0, where=(df['Spread_2s10s'] >= 0), 
                     color='#27ae60', alpha=0.3, label='Normal Curve')
    ax2.fill_between(df.index, df['Spread_2s10s'], 0, where=(df['Spread_2s10s'] < 0), 
                     color='#c0392b', alpha=0.4, label='Inverted (Recession Risk)')
    
    ax2.axhline(0, color='black', lw=1, linestyle='-')
    ax2.set_ylabel('10Y-2Y Spread (%)', fontsize=11, fontweight='bold')
    ax2.set_xlabel('Date', fontsize=11)
    ax2.legend(loc='lower right')
    ax2.grid(True, alpha=0.2)

    plt.tight_layout()
    plt.show()

plot_comprehensive_economic_analysis(final_df)

def plot_with_event_lines(df):
    plt.figure(figsize=(12, 6))
    
    # 1. Plot the Market Rates
    plt.plot(df.index, df['2Y_Yield'], label='2Y Gilt Yield (Market)', color='#007acc', lw=2)
    plt.plot(df.index, df['Base_Rate'], label='BoE Base Rate (Policy)', color='#333333', linestyle='--', alpha=0.7)
    
    # 2. Add Vertical Event Lines
    # Event A: The last 25bps cut
    last_cut = pd.to_datetime('2025-12-18')
    plt.axvline(x=last_cut, color='green', linestyle=':', lw=2, label='Dec 2025 Cut (3.75%)')
    
    # Event B: Upcoming February meeting
    next_meeting = pd.to_datetime('2026-02-05')
    plt.axvline(x=next_meeting, color='red', linestyle=':', lw=2, label='Feb 2026 Meeting (Upcoming)')

    # 3. Highlight the "Gap" (The Market Prediction)
    # If the blue line is BELOW the black dashed line, the market is betting on a cut.
    # If the blue line is ABOVE, the market is betting on a pause/hike.
    
    plt.title('UK 2Y Gilt vs. BoE Policy: Betting on February 5th', fontsize=14)
    plt.ylabel('Yield / Rate (%)')
    plt.legend()
    plt.grid(alpha=0.3)
    plt.show()

plot_with_event_lines(final_df)

def generate_trade_signal(df):
    latest = df.iloc[-1]
    
    current_yield = latest['2Y_Yield']
    base_rate = latest['Base_Rate']
    cpi = latest['CPI_YoY']
    
    #'Market Pricing'
    # If gap is negative, market expects cuts. If positive, market expects hikes.
    policy_gap = current_yield - base_rate
    
    print(f"--- Tactical Rates View: {df.index[-1].date()} ---")
    print(f"Current Base Rate: {base_rate}% | 2Y Gilt Yield: {current_yield}%")
    print(f"Market 'Pricing In': {policy_gap:.2f}% vs the Base Rate")
    print("-" * 40)

    # Very Simple Logic
    if policy_gap < -0.25:
        view = "Saturated. The market already expects a cut. High risk to 'Buying' here."
    elif policy_gap > 0.10:
        view = "Bearish/Selling. Market is too hawkish. Opportunity to 'Buy' if CPI cools."
    elif -0.10 <= policy_gap <= 0.10:
        view = "Neutral. The market is perfectly aligned with the BoE."
    else:
        view = "Tactical Buy. Market is doubting a cut that the macro data (CPI) supports."
        
    return view

recommendation = generate_trade_signal(final_df)
print(f"Recommendation: {recommendation}")

def print_market_narrative(df):
    latest = df.iloc[-1]
    
    # Extract values for the report
    yield_2y = latest['2Y_Yield']
    base_rate = latest['Base_Rate']
    cpi = latest['CPI_YoY']
    gap = yield_2y - base_rate
    
    print("="*60)
    print("         UK GILTS WEEKLY         ")
    print(f"                Date: {df.index[-1].date()}                ")
    print("="*60)
    
    print(f"\n1. MARKET PRICING")
    print(f"   The 2Y Gilt yield is currently {yield_2y:.3f}%, trading at a ")
    print(f"   spread of {gap*100:.1f}bps to the {base_rate}% Base Rate. This indicates")
    print(f"   the market has effectively 'priced out' a near-term cut.")
    
    print(f"\n2. MACRO CONTEXT")
    print(f"   With CPI currently at {cpi}%, the real policy rate remains thin.")
    print(f"   The lack of a 'Policy Gap' suggests the market is in a")
    print(f"   'wait-and-see' mode ahead of the next BoE announcement.")
    
    print(f"\n3. THE 'CONVEXITY' PLAY (Binary Outcomes)")
    print(f"    SCENARIO A (Dovish Surprise): If the BoE cuts, 2Y yields")
    print(f"     likely reprice toward 3.50% (Significant 'Long' Profit).")
    print(f"    SCENARIO B (Hawkish Hold): If the BoE signals stickier")
    print(f"     inflation, 2Y yields likely spike toward 3.90%+")
    print(f"     (Significant 'Short' Profit).")
    
    print("\n" + "="*60)

# Run the report
